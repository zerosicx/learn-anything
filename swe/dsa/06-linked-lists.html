Linked Lists

Dynamic data structures built from nodes and pointers — trading random access for efficient insertion and deletion

TypeScript Pointers

What is a Linked List?

A linked list is a linear data structure where each element (called a node) contains a value and a pointer to the next node. Unlike arrays, linked list nodes aren't stored contiguously in memory — each node can live anywhere, connected only by its pointer.

This pointer-based structure means insertion and deletion at any known position is O(1) — you just reroute pointers. But accessing the i-th element requires walking from the head, making random access O(n).

Operation Array Linked List
Access by index O(1) O(n)
Search O(n) O(n)
Insert at head O(n) O(1)
Insert at tail O(1)* O(1)**

ListNode

class ListNode<T> {
  constructor(
    public val: T,
    public next: ListNode<T> | null = null
  ) {}
}

LinkedList class

class LinkedList<T> {
  private head: ListNode<T> | null = null;
  private size = 0;

  append(val: T): void {
    const node = new ListNode(val);
    if (!this.head) {
      this.head = node;
    } else {
      let curr = this.head;
      while (curr.next) curr = curr.next;
      curr.next = node;
    }
    this.size++;
  }

  prepend(val: T): void {
    this.head = new ListNode(val, this.head);
    this.size++;
  }

  delete(val: T): boolean {
    if (!this.head) return false;

    if (this.head.val === val) {
      this.head = this.head.next;
      this.size--;
      return true;
    }

    let curr = this.head;
    while (curr.next) {
      if (curr.next.val === val) {
        curr.next = curr.next.next;
        this.size--;
        return true;
      }
      curr = curr.next;
    }
    return false;
  }

  find(val: T): ListNode<T> | null {
    let curr = this.head;
    while (curr) {
      if (curr.val === val) return curr;
      curr = curr.next;
    }
    return null;
  }

  toArray(): T[] {
    const result: T[] = [];
    let curr = this.head;
    while (curr) {
      result.push(curr.val);
      curr = curr.next;
    }
    return result;
  }

  get length(): number { return this.size; }
}

Doubly Linked List

A doubly linked list adds a prev pointer to each node, enabling O(1) deletion without needing a reference to the preceding node and efficient traversal in both directions. The trade-off is extra memory per node.

class DoublyNode<T> {
  constructor(
    public val: T,
    public prev: DoublyNode<T> | null = null,
    public next: DoublyNode<T> | null = null
  ) {}
}

class DoublyLinkedList<T> {
  private head: DoublyNode<T> | null = null;
  private tail: DoublyNode<T> | null = null;
  private size = 0;

  append(val: T): DoublyNode<T> {
    const node = new DoublyNode(val);
    if (!this.tail) {
      this.head = this.tail = node;
    } else {
      node.prev = this.tail;
      this.tail.next = node;
      this.tail = node;
    }
    this.size++;
    return node;
  }

  prepend(val: T): DoublyNode<T> {
    const node = new DoublyNode(val);
    if (!this.head) {
      this.head = this.tail = node;
    } else {
      node.next = this.head;
      this.head.prev = node;
      this.head = node;
    }
    this.size++;
    return node;
  }

  remove(node: DoublyNode<T>): void {
    if (node.prev) node.prev.next = node.next;
    else this.head = node.next;

    if (node.next) node.next.prev = node.prev;
    else this.tail = node.prev;

    node.prev = node.next = null;
    this.size--;
  }

  moveToFront(node: DoublyNode<T>): void {
    this.remove(node);
    node.next = this.head;
    if (this.head) this.head.prev = node;
    this.head = node;
    if (!this.tail) this.tail = node;
    this.size++;
  }

  get headNode(): DoublyNode<T> | null { return this.head; }
  get tailNode(): DoublyNode<T> | null { return this.tail; }
  get length(): number { return this.size; }
}

Reverse a Linked List

Reversing a linked list is the quintessential pointer manipulation problem. There are two approaches: iterative (constant space) and recursive (O(n) call stack).

Iterative

function reverseList(head: ListNode<number> | null): ListNode<number> | null {
  let prev: ListNode<number> | null = null;
  let curr = head;

  while (curr) {
    const next = curr.next;  // Save next before we overwrite it
    curr.next = prev;         // Reverse the pointer
    prev = curr;              // Advance prev
    curr = next;              // Advance curr
  }

  return prev; // prev is the new head
}

Recursive

function reverseListRecursive(head: ListNode<number> | null): ListNode<number> | null {
  if (!head || !head.next) return head;

  const newHead = reverseListRecursive(head.next);

  // head.next is the last node of the reversed sublist
  // Make it point back to head
  head.next.next = head;
  head.next = null;

  return newHead;
}

Pointer Manipulation Pitfall

Always save curr.next before overwriting it. Once you set curr.next = prev, the original next reference is lost forever. This is the single most common linked list bug.

Detect a Cycle — Floyd's Algorithm

How do you detect if a linked list has a cycle? You can't just walk to the end and check for null — if there's a cycle, you'll loop forever. Floyd's cycle detection uses two pointers moving at different speeds: slow (1 step) and fast (2 steps). If there's a cycle, they will eventually meet. If there isn't, fast will reach null.

function hasCycle(head: ListNode<number> | null): boolean {
  let slow = head;
  let fast = head;

  while (fast && fast.next) {
    slow = slow!.next;
    fast = fast.next.next;

    if (slow === fast) return true;
  }

  return false;
}
Why does this work? Think of the cycle as a circular track. Once both pointers enter the cycle, the fast pointer closes the gap by 1 position each step. Since the gap decreases by 1 every iteration, they must eventually collide — guaranteed within one full revolution of the cycle.

Merge Two Sorted Lists

Given two sorted linked lists, merge them into one sorted list. Use a dummy head to simplify edge cases — you don't need to special-case which list provides the first node.

function mergeTwoLists(
  l1: ListNode<number> | null,
  l2: ListNode<number> | null
): ListNode<number> | null {
  const dummy = new ListNode(0);
  let tail = dummy;

  while (l1 && l2) {
    if (l1.val <= l2.val) {
      tail.next = l1;
      l1 = l1.next;
    } else {
      tail.next = l2;
      l2 = l2.next;
    }
    tail = tail.next;
  }

  tail.next = l1 ?? l2;
  return dummy.next;
}

The Dummy Head Pattern

A dummy (sentinel) node at the start of a list eliminates null-checking for the first insertion. The real list starts at dummy.next. This pattern appears in almost every linked list construction problem.

LRU Cache

An LRU (Least Recently Used) cache evicts the least recently accessed entry when full. It combines two data structures: a doubly linked list for O(1) ordering (most-recent at head, least-recent at tail) and a HashMap for O(1) key lookup.

class LRUNode {
  constructor(
    public key: number,
    public val: number,
    public prev: LRUNode | null = null,
    public next: LRUNode | null = null
  ) {}
}

class LRUCache {
  private capacity: number;
  private map = new Map<number, LRUNode>();
  // Sentinel nodes avoid null checks
  private head = new LRUNode(0, 0);
  private tail = new LRUNode(0, 0);

  constructor(capacity: number) {
    this.capacity = capacity;
    this.head.next = this.tail;
    this.tail.prev = this.head;
  }

  private remove(node: LRUNode): void {
    node.prev!.next = node.next;
    node.next!.prev = node.prev;
  }

  private insertAfterHead(node: LRUNode): void {
    node.next = this.head.next;
    node.prev = this.head;
    this.head.next!.prev = node;
    this.head.next = node;
  }

  get(key: number): number {
    const node = this.map.get(key);
    if (!node) return -1;

    this.remove(node);
    this.insertAfterHead(node);
    return node.val;
  }

  put(key: number, value: number): void {
    if (this.map.has(key)) {
      this.remove(this.map.get(key)!);
    }

    const node = new LRUNode(key, value);
    this.map.set(key, node);
    this.insertAfterHead(node);

    if (this.map.size > this.capacity) {
      const lru = this.tail.prev!;
      this.remove(lru);
      this.map.delete(lru.key);
    }
  }
}
Why doubly linked list + HashMap? The HashMap gives O(1) lookup by key. The doubly linked list gives O(1) removal and insertion to track recency order. Neither data structure alone can do both — arrays have O(n) removal, and hash maps don't maintain order.
